### 至少实现三种排序算法（从小到大）

1. 冒泡：

```javascript
for (let i = 0; i < test.length; i++) {
    for (let j = 0; j < test.length-i-1; j++) {
        if (test[j] < test[j+1]) {
            t = test[j+1];
            test[j+1] = test[j];
            test[j] = t
        }
    }
}
```
2. 快速排序

```javascript
let test = [1, 2, 4, 3, 5];

function sort(test) {
    console.log(test.length);
    let num = Math.floor(test.length / 2)
    console.log(num);
    let value = test.splice(num, 1)[0]
    console.log(value);
    let left = [], right = [];

    for (let i = 0; i < test.length; i++) {
        console.log(test[i]);
        if (test[i] < value) {
            left.push(test[i])
        } else {
            right.push(test[i])
        }
    }

    return sort(left).concat([value], sort(right))
}
```
